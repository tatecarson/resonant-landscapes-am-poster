% the main text of your poster goes here
\makemain{
    % you can have 1 or 2 columns
    \raggedcolumns
    \begin{multicols}{2}
        \section{Introduction}
        \begin{compactitem}
            \item Objective: Merge technology and nature.
            \item Concept: Overlay soundscape recordings from South Dakota state parks onto the DSU campus.
        \end{compactitem}

        \section{Key Features and Technical Implementation}
        \begin{compactitem}
            \item \textbf{Ambisonic Audio:} 2nd-order Ambisonic audio, captured with Core Audio OctoMic.
            \item \textbf{Body-oriented Tracking:} Leverages smartphone sensors to allow users to alter their listening perspective by turning their bodies.
            \item \textbf{Interaction Design:} Users navigate DSU campus, encountering various listening spots with dynamic soundscapes.
            \item \textbf{Platform Architecture:} Built with React, Tailwind CSS, Resonance Audio SDK, Omnitone, Turf.js, and Three.js.
        \end{compactitem}

        % this determines where your columns will be separated
        \columnbreak

        \section{Ecological and Cultural Significance}
        \begin{compactitem}
            \item Combines natural and urban soundscapes to foster ecological awareness.
            \item Maps environmental soundscapes onto physical locations, creating a "hybrid place".
        \end{compactitem}

    \end{multicols}
