\makefinding{
    \textbf{Resonant Landscapes} integrates \textbf{ambisonic audio} and \textbf{GPS technology} to create immersive natural soundscapes in urban environments.
}

\makemain{
    \raggedcolumns
    \begin{multicols}{2}
        \section{Introduction}
        \begin{compactitem}
            \item Overlay South Dakota state park soundscapes onto DSU campus.
            \item Web-based application using frugal innovation principles.
        \end{compactitem}

        \section{Key Features}
        \begin{compactitem}
            \item 2nd-order ambisonic audio for immersive experience.
            \item GPS integration maps state parks to campus locations.
            \item Body-oriented tracking using smartphone sensors.
            \item Dynamic soundscapes based on user proximity.
        \end{compactitem}

        \section{Technical Implementation}
        \begin{compactitem}
            \item Core Audio OctoMic for 8-channel audio capture.
            \item Web technologies: React, Tailwind CSS, Resonance Audio SDK.
            \item Utilizes smartphone GPS, gyroscope, and accelerometer.
        \end{compactitem}

        \columnbreak

        \section{User Experience}
        \begin{compactitem}
            \item Interactive campus map with listening spots.
            \item Audio playback within 15-meter radius.
            \item Body-oriented tracking at spot epicenters.
        \end{compactitem}

        \section{Significance}
        \begin{compactitem}
            \item Creates "hybrid place" of natural and urban soundscapes.
            \item Promotes ecological awareness and attentive listening.
        \end{compactitem}

        \section{Future Work}
        \begin{compactitem}
            \item Enhance user experience through iterative design.
            \item Expand soundscape database for diverse ecosystems.
        \end{compactitem}
    \end{multicols}
}
